\section{Формулы}

\textbf{Inline.} Пусть $f : \mathbb{R}^n \to \mathbb{R}$ --- выпуклая
и дифференцируемая; градиентный шаг с темпом $\eta > 0$ --- это
$x_{k+1} = x_k - \eta \nabla f(x_k)$.

\textbf{Display.} Классическое неравенство Коши--Шварца:
\begin{equation}
  \bigl|\langle u, v\rangle\bigr| \;\leq\; \|u\|\cdot\|v\|,\qquad u,v\in H.
  \label{eq:cauchy}
\end{equation}

Внутри доказательства удобно ссылаться на~\eqref{eq:cauchy} через
\texttt{\textbackslash eqref}.

% ---- Теорема и определение ------------------------------------------------
\begin{definition}[Онтологическая тройка]
\emph{Онтологическая тройка} --- упорядоченная тройка
$\langle s, p, o \rangle$, где $s, o$ --- сущности, $p$ --- типизированное
отношение.
\end{definition}

\begin{theorem}[Разложение графа знаний]
\label{thm:kg}
Любой конечный граф знаний $\mathcal{K}$ представим как мультимножество
$\{\langle s_i, p_i, o_i\rangle\}_{i=1}^{N}$, где $N$ --- число рёбер.
\end{theorem}

\begin{proof}
Индукция по $N$. Каждое ребро графа знаний по определению типизировано.
Остаётся собрать их все в мультимножество; порядок игнорируем.
\end{proof}

% ---- Правила подачи -------------------------------------------------------
\section{Как подать на учебную конференцию AI4Math-YSDA 2026}

\begin{enumerate}
  \item \textbf{Регистрация на OpenReview.}
    Откройте \url{https://openreview.net}, создайте профиль с
    академическим адресом. Укажите affiliation (ШАД / ваш университет).
  \item \textbf{Оформление работы.} Используйте этот шаблон.
    Заменяйте текст в \texttt{example\_paper.tex} на свой.
    \texttt{\textbackslash usepackage\{ai4math\_ysda2026\}} --- не трогайте.
  \item \textbf{Сдача статьи.} На странице учебного venue (ссылка
    будет разослана преподавателем до 1~мая 2026~г.) --- нажмите
    <<New Submission>>, загрузите собранный PDF. Deadline:
    \textbf{5~мая 2026}.
  \item \textbf{Ожидайте рецензий.} Ваша работа попадёт в bidding
    (6~мая), затем в assignment (7~мая). С 7~по 12~мая --- период
    рецензирования. Параллельно вы сами пишете 3~рецензии (см. день~7
    курса).
  \item \textbf{Rebuttal.} 12--13~мая --- период ответа авторам.
    Отвечайте по существу, не переделывая статью.
  \item \textbf{Финальные оценки.} 14~мая --- обсуждение reviewers~+
    AC. 15~мая --- нотификации.
\end{enumerate}

\textbf{Ключевые термины.} Area Chair (AC) --- ответственный за
модерацию рецензий и принятие решения; Desk Reject --- немедленный
отказ до рецензирования за формальные нарушения; PC --- Programme
Chair, утверждает решения AC.
